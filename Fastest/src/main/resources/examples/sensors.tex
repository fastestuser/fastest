\begin{schema}{aa}
a1 : \nat \pfun \num \\
a2 : \nat \fun \num \\
a3: \power \num
\end{schema}

\begin{zed}
[sENSOR] \\
FT ::= ft1 | ft2 | ft3 | ft4 | ft5 | ft6
\end{zed}

\begin{schema}{B}
sen, sen' : sENSOR \\
%x,x' : FT \fun \num \\
%y,y' : FT \pfun \num \\
%p: \power aa \\
%k: \power (SENSOR \cross \num)\\
%num1, num2 : \num \\
a,b : aa
\end{schema}

\begin{schema}{C}
b1,b2,b2' : B
\where
b1 \neq b2
\end{schema}
